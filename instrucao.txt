python -m venv nome do projeto    (cria a maquina virtual)
venv\scripts\activate             (ativa a maquina virtual)
pip install django                (instala o django na maquina virtual)
django-admin                        (adcionar o admin)
django-admin startproject setup .    (inicia o projeto com nome setup)
django-admin startapp cadastro       (inicia o app com nome cadastro)
python manage.py runserver              (roda o runserver)
python -m pip install Pillow            ()
python manage.py migrate                (migra as migrações que ja estão prontas)
python manage.py makemigrations         (prepara a migração)
python manage.py createsuperuser        (criar o usuario e cadastrar a senha)

git config --global user.email "email@example.com"  (adcionar o email)
git config --global user.name "Username"
git push --set-upstream origin master   (prepara o push)
git add . (adciona todos os arquivos da pasta)
git commit -m "comentário"
